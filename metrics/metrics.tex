% From: tex.stackexchange.com/questions/409930
% Shows current size with \currentsize which can be stored to \fsize
% with \setfsize. The length \size can then be printed with \prntlen
\makeatletter
\newcommand{\currentfsize}{\f@size pt}
\makeatother
\newdimen\fsize
\newcommand{\setfsize}{\setlength{\fsize}{\currentfsize}}

% Strip unit pt from font sizes to round actual numbers
\makeatletter
\newcommand*{\stripunit}[1]{\strip@pt#1}
\makeatother
